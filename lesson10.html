        for row in reader:
            # –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π –≤ —á–∏—Å–ª–æ–≤—ã–µ

            planets.append(row)

    # –ê–Ω–∞–ª–∏–∑ –¥–∞–Ω–Ω—ã—Ö
    total_planets = len(planets)

    largest_planet = max(planets, key=lambda p: p["–î–∏–∞–º–µ—Ç—Ä (–∫–º)"])

    print(f"–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–ª–∞–Ω–µ—Ç: {total_planets}")
    print(f"–û–±—â–µ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ø—É—Ç–Ω–∏–∫–æ–≤: {total_moons}")
    print(f"–°—Ä–µ–¥–Ω–∏–π –¥–∏–∞–º–µ—Ç—Ä –ø–ª–∞–Ω–µ—Ç: {avg_diameter:.2f} –∫–º")
    print(f"–°–∞–º–∞—è –±–æ–ª—å—à–∞—è –ø–ª–∞–Ω–µ—Ç–∞: {largest_planet['–ù–∞–∑–≤–∞–Ω–∏–µ']} (–¥–∏–∞–º–µ—Ç—Ä: {largest_planet['–î–∏–∞–º–µ—Ç—Ä (–∫–º)']} –∫–º)")
